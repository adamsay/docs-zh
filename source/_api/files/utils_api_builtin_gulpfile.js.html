                function wrapScope () {
                    var header = new Buffer("(function () {\n");
                    var footer = new Buffer("})();\n");
                    return es.through(function (file) {
                        file.contents = Buffer.concat([header, file.contents, footer]);
                        this.emit('data', file);
                    });
                }
                
                var task_copy_deps = [];
                var task_min_deps = [];
                var task_dev_deps = [];
                var plugin_watchers = [];
                
                var task_plugin = function ( name ) {
                    var basePath = name + '/';
                    var elementsPath = name + '/editor/elements/';
                    var extPath = name + '/editor/ext/';
                    var destBinPath = name + '/bin/';
                    var destBinTmpPath = name + '/bin/tmp/';
                
                    var task_copy = 'plugin-' + name + '-copy';
                    var task_copy_ext = 'plugin-' + name + '-copy-ext';
                    var task_copy_res = 'plugin-' + name + '-copy-res';
                    var task_copy_html = 'plugin-' + name + '-copy-html';
                
                    var task_css = 'plugin-' + name + '-css';
                    var task_styl = 'plugin-' + name + '-styl';
                    var task_js = 'plugin-' + name + '-js';
                    var task_js_dev = 'plugin-' + name + '-js-dev';
                    var task_runtime_js = 'plugin-' + name + '-runtime-js';
                    var task_html = 'plugin-' + name + '-html';
                    var task_html_dev = 'plugin-' + name + '-html-dev';
                
                    task_copy_deps.push(task_copy_html, task_copy_res);
                    task_min_deps.push(task_html);
                    task_min_deps.push(task_runtime_js);
                    task_dev_deps.push(task_html_dev);
                    task_dev_deps.push(task_runtime_js);
                
                    var js_files = elementsPath + '**/*.js';
                    var runtime_js_files = [basePath + '**/*.js',
                                            basePath + 'src/**/*.js',
                                            '!' + basePath + '**/{Editor,editor}/**',
                                            '!' + destBinPath + '**'];
                
                    var watcher = {
                        css: { files: elementsPath + '**/*.css', tasks: [task_css, task_html_dev] },
                        styl: { files: elementsPath + '**/*.styl', tasks: [task_styl, task_html_dev] },
                        js: { files: js_files, tasks: [task_html_dev] },
                        runtime_js: { files: runtime_js_files, tasks: [task_runtime_js] },
                        html: { files: elementsPath + '**/*.html', tasks: [task_html_dev] },
                        res: { files: [elementsPath + '**/*.jpg', elementsPath + '**/*.png'], tasks: [task_copy_res] }
                    };
                    plugin_watchers.push(watcher);
                
                    // copy
                    gulp.task( task_copy, function() {
                        return gulp.src( [
                            basePath + '**/*',
                            '!' + elementsPath,
                            '!' + elementsPath + "**/*",
                            '!' + extPath,
                            '!' + extPath + "**/*",
                            '!' + basePath + "bin",
                            '!' + basePath + "bin/**/*",
                        ], {base: basePath})
                        .pipe(gulp.dest(destBinPath))
                        ;
                    });
                
                    gulp.task( task_copy_ext, function() {
                        return gulp.src( [
                            extPath + '**/*',
                        ])
                        .pipe(gulp.dest(destBinPath + "ext/"))
                        ;
                    });
                
                    gulp.task( task_copy_html, function() {
                        return gulp.src( [
                            elementsPath + '**/*.html',
                        ], {base: elementsPath})
                        .pipe(gulp.dest(destBinTmpPath))
                        ;
                    });
                
                    gulp.task( task_copy_res, function() {
                        return gulp.src( [
                            elementsPath + '**/*.jpg',
                            elementsPath + '**/*.png',
                        ], {base: elementsPath})
                        .pipe(gulp.dest(destBinPath))
                        ;
                    });
                
                    // styl
                    gulp.task( task_styl, function() {
                        return gulp.src(elementsPath + '**/*.styl', {base: elementsPath})
                        .pipe(stylus({
                            compress: true,
                            include: 'src'
                        }))
                        .pipe(gulp.dest(destBinTmpPath));
                    });
                
                    // css
                    gulp.task( task_css, function() {
                        return gulp.src(elementsPath + '**/*.css', {base: elementsPath})
                        .pipe(minifyCSS())
                        .pipe(gulp.dest(destBinTmpPath));
                    });
                
                    // js
                    gulp.task(task_js, function() {
                        var uglify = require('gulp-uglifyjs');
                
                        return gulp.src(js_files, {base: elementsPath})
                        .pipe(wrapScope())
                        .pipe(jshint({
                            multistr: true,
                            smarttabs: false,
                            loopfunc: true,
                            esnext: true,
                        }))
                        .pipe(jshint.reporter(stylish))
                        .pipe(uglify({
                            compress: {
                                dead_code: false,
                                unused: false
                            }
                        }))
                        .pipe(gulp.dest(destBinTmpPath))
                        ;
                    });
                
                    gulp.task(task_js_dev, function() {
                        return gulp.src(js_files, {base: elementsPath})
                        .pipe(wrapScope())
                        .pipe(jshint({
                            multistr: true,
                            smarttabs: false,
                            loopfunc: true,
                            esnext: true,
                        }))
                        .pipe(jshint.reporter(stylish))
                        .pipe(gulp.dest(destBinTmpPath))
                        ;
                    });
                
                    gulp.task(task_runtime_js, function() {
                        return gulp.src(runtime_js_files, {base: basePath})
                        .pipe(jshint({
                            multistr: true,
                            smarttabs: false,
                            loopfunc: true,
                            esnext: true,
                        }))
                        .pipe(jshint.reporter(stylish))
                        .pipe(gulp.dest(destBinPath))
                        ;
                    });
                
                    // html
                    var build_html = function (strip) {
                        return function () {
                            return gulp.src(destBinTmpPath + '/' + name + '.html')
                            .pipe(vulcanize({
                                dest: destBinTmpPath,
                                inline: true,
                                strip: strip,
                            }))
                            .pipe(gulp.dest(destBinPath + "/editor/"))
                            ;
                        };
                    };
                
                    gulp.task(task_html, [ task_copy, task_copy_ext, task_copy_html, task_copy_res, task_styl, task_css, task_js ], build_html(true));
                    gulp.task(task_html_dev, [ task_copy, task_copy_ext, task_copy_html, task_copy_res, task_styl, task_css, task_js_dev ], build_html(false));
                };

                gulp.task('watch', function() {

                        gulp.watch( watcher.js.files, watcher.js.tasks ).on ( 'error', gutil.log );
                        gulp.watch( watcher.runtime_js.files, watcher.runtime_js.tasks ).on ( 'error', gutil.log );
                        gulp.watch( watcher.html.files, watcher.html.tasks ).on ( 'error', gutil.log );
                        gulp.watch( watcher.res.files, watcher.res.tasks ).on ( 'error', gutil.log );
                    }
                });

                    var src = ['./**/*.js', '!./*/bin/**/*', '!./*/editor/ext/**/*'];

                            return;

                            .pipe(gulp.dest(dest))
                            .on('end', done);
                    });
